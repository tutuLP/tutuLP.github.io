# 🧱 三、Pinia（状态层）

```
// stores/authModal.ts
import { defineStore } from "pinia";

export type AuthMode = "login" | "register";

export const useAuthModalStore = defineStore("authModal", {
  state: () => ({
    visible: false,
    mode: "login" as AuthMode,

    // Promise控制
    resolver: null as null | ((val: any) => void)
  }),

  actions: {
    open(mode: AuthMode = "login") {
      this.mode = mode;
      this.visible = true;
    },

    close() {
      this.visible = false;
    },

    // 登录成功时调用
    resolve(data: any) {
      this.resolver?.(data);
      this.resolver = null;
      this.close();
    }
  }
});
```
