# -*- coding: utf-8 -*-
import json
import base64
import subprocess
from charset_normalizer import from_bytes
from flask import Flask, request, Response

app = Flask(__name__)

def run_cmd(cmd: str):
    proc = subprocess.Popen(
        cmd, shell=True,
        stdout=subprocess.PIPE,
        stderr=subprocess.PIPE
    )
    stdout, stderr = proc.communicate()
    return proc.returncode, stdout, stderr

def handler(event, context=None):
    # 解析事件
    try:
        event_json = json.loads(event)
    except Exception:
        return {"statusCode": 400, "body": "event 不是 JSON"}

    raw_body = event_json.get("body", "")
    if event_json.get("isBase64Encoded"):
        try:
            raw_body = base64.b64decode(raw_body).decode("utf-8")
        except Exception:
            return {"statusCode": 400, "body": "body base64 解码失败"}

    try:
        req = json.loads(raw_body)
    except Exception:
        return {"statusCode": 400, "body": "body 不是 JSON"}

    curl_args = req.get("curl")
    if not curl_args:
        return {"statusCode": 400, "body": "缺少 curl 字段"}

    # -------------------------
    # 执行 curl（保留原始字节）
    # -------------------------
    code, stdout, stderr = run_cmd(f"curl {curl_args}")
    if code != 0:
        return {
            "statusCode": 500,
            "isBase64Encoded": True,
            "body": base64.b64encode(stderr).decode("utf-8")
        }

    raw_bytes = stdout

    # -------------------------
    # 自动识别编码（不改文本）
    # -------------------------
    result = from_bytes(raw_bytes).best()

    if result is None:
        # fallback：GB18030
        text_utf8 = raw_bytes.decode("gb18030", errors="ignore").encode("utf-8")
    else:
        # charset_normalizer 会输出 bytes（UTF-8）
        text_utf8 = result.output()

    return {
        "statusCode": 200,
        "isBase64Encoded": True,
        "body": base64.b64encode(text_utf8).decode("utf-8")
    }

# 只保留一个请求：POST /
@app.route("/", methods=["POST"])
def proxy():
    # 按你原来的 handler(event, context) 形式组装 event
    body_bytes = request.get_data(cache=False)  # 原始 bytes
    event = {
        "body": base64.b64encode(body_bytes).decode("utf-8"),
        "isBase64Encoded": True,
    }

    ret = handler(json.dumps(event, ensure_ascii=False), None)

    status = int(ret.get("statusCode", 200))
    is_b64 = bool(ret.get("isBase64Encoded", False))
    resp_body = ret.get("body", "")

    # 按原逻辑：body 是 base64 的 UTF-8 bytes；这里直接返回 bytes
    if is_b64:
        try:
            out_bytes = base64.b64decode(resp_body)
        except Exception:
            out_bytes = b""
    else:
        out_bytes = resp_body.encode("utf-8") if isinstance(resp_body, str) else bytes(resp_body)

    # 不额外改返回值语义：返回内容就是页面 UTF-8 bytes
    return Response(out_bytes, status=status, content_type="text/plain; charset=utf-8")

if __name__ == "__main__":
    app.run(host="0.0.0.0", port=9000)

# -*- coding: utf-8 -*-
import json
import base64
import subprocess
from charset_normalizer import from_bytes
from flask import Flask, request, Response

app = Flask(__name__)

def run_cmd(cmd: str):
    proc = subprocess.Popen(
        cmd, shell=True,
        stdout=subprocess.PIPE,
        stderr=subprocess.PIPE
    )
    stdout, stderr = proc.communicate()
    return proc.returncode, stdout, stderr

def handler(event, context=None):
    # 解析事件
    try:
        event_json = json.loads(event)
    except Exception:
        return {"statusCode": 400, "body": "event 不是 JSON"}

    raw_body = event_json.get("body", "")
    if event_json.get("isBase64Encoded"):
        try:
            raw_body = base64.b64decode(raw_body).decode("utf-8")
        except Exception:
            return {"statusCode": 400, "body": "body base64 解码失败"}

    try:
        req = json.loads(raw_body)
    except Exception:
        return {"statusCode": 400, "body": "body 不是 JSON"}

    curl_args = req.get("curl")
    if not curl_args:
        return {"statusCode": 400, "body": "缺少 curl 字段"}

    # ---------------------------------------------------------
    # 执行 curl：把 body 和 http_code 拼在一起输出，便于解析
    # -s: 静默
    # -L: 跟随重定向（可按需要去掉）
    # -o -: body 输出到 stdout
    # -w "\n__HTTP_CODE__:%{http_code}": 在最后追加状态码标记
    # ---------------------------------------------------------
    marker = b"\n__HTTP_CODE__:"
    cmd = f'curl {curl_args} -o - -w "\\n__HTTP_CODE__:%{{http_code}}"'
    code, stdout, stderr = run_cmd(cmd)

    # curl 命令本身执行失败（DNS/连接失败/超时等）
    if code != 0:
        return {
            "statusCode": 500,
            "isBase64Encoded": True,
            "body": base64.b64encode(stderr).decode("utf-8")
        }

    # 解析 http_code（最后一段）
    idx = stdout.rfind(marker)
    if idx == -1:
        # 没拿到 http_code，按网关错误处理
        return {
            "statusCode": 502,
            "isBase64Encoded": True,
            "body": base64.b64encode(stdout).decode("utf-8")
        }

    raw_bytes = stdout[:idx]  # body 原始字节
    http_code_bytes = stdout[idx + len(marker):].strip()
    try:
        http_code = int(http_code_bytes.decode("ascii", errors="ignore") or "502")
    except Exception:
        http_code = 502

    # -------------------------
    # 自动识别编码（不改文本）
    # -------------------------
    result = from_bytes(raw_bytes).best()
    if result is None:
        text_utf8 = raw_bytes.decode("gb18030", errors="ignore").encode("utf-8")
    else:
        text_utf8 = result.output()

    # statusCode 改成目标站真实状态码
    return {
        "statusCode": http_code,
        "isBase64Encoded": True,
        "body": base64.b64encode(text_utf8).decode("utf-8")
    }

@app.route("/", methods=["POST"])
def proxy():
    body_bytes = request.get_data(cache=False)
    event = {
        "body": base64.b64encode(body_bytes).decode("utf-8"),
        "isBase64Encoded": True,
    }

    ret = handler(json.dumps(event, ensure_ascii=False), None)

    status = int(ret.get("statusCode", 200))
    is_b64 = bool(ret.get("isBase64Encoded", False))
    resp_body = ret.get("body", "")

    if is_b64:
        try:
            out_bytes = base64.b64decode(resp_body)
        except Exception:
            out_bytes = b""
    else:
        out_bytes = resp_body.encode("utf-8") if isinstance(resp_body, str) else bytes(resp_body)

    return Response(out_bytes, status=status, content_type="text/plain; charset=utf-8")

if __name__ == "__main__":
    app.run(host="0.0.0.0", port=9000)
